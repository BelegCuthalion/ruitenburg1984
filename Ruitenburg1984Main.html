Corollary rui_1_4´: forall A b n, (Bound [] A b) -> cardinal form b n ->
                              forall i v,
                                let v´ := S v in
                                fresh_f_p v´ A ->
                                          let G´ := [f_p A i ; sub (s_p tt) A] in
                                          let G´´ := (f_p A (2*n) ->> (var v´))::G´ in
                 G´´ |-- (sub (s_p (var v´)) A <<->> sub (s_p tt) A) & (sub (s_p tt) A ->> (var v´)) \/
                 G´´ |-- sub (s_p (var v´)) A <<->> (var v´) \/
                 G´´ |-- sub (s_p (var v´)) A.
